空 演示乘(整数 二维数组[][2], 整数 一维数组[]){
    如果(二维数组[0][0] 恒等于 1){
        打印("%d\n",一维数组[0]);
        返回;
    }否则{
        一维数组[0] 等于 一维数组[0] 乘 二维数组[0][0];
        二维数组[0][0] 等于 二维数组[0][0] 减 1;
        演示乘(二维数组, 一维数组);
        返回;
    }
    返回;
}
